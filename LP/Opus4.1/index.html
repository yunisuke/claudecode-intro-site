def find_primes(n):
    """Find all prime numbers up to n using Sieve of Eratosthenes"""
    if n < 2:
        return []
    
    sieve = [True] * (n + 1)
    sieve[0] = sieve[1] = False
    
    for i in range(2, int(n**0.5) + 1):
        if sieve[i]:
            for j in range(i*i, n + 1, i):
                sieve[j] = False
    
    return [i for i in range(n + 1) if sieve[i]]

const memoize = fn => {
    const cache = {};
    return (...args) => {
        const key = JSON.stringify(args);
        return cache[key] || (cache[key] = fn(...args));
    };
};
Claude Code:

This is a memoization higher-order function that optimizes performance by caching function results.

How it works:

  • Creates a cache object to store results
  • Returns a new function that checks cache before executing
  • Uses JSON.stringify to create unique keys from arguments
  • Returns cached result if available, otherwise computes and caches
